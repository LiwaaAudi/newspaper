# basic_blog
A basic blog web app using django

# Start web app

    $ pipenv install
    $ pipenv shell
    (blog) $ python manage.py migrate
    (blog) $ python manage.py runserver
    # Load the site at http://127.0.0.1:8000'
    
 
# Run Test


    (blog) $ python manage.py test
